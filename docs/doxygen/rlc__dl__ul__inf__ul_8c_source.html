437 S16 rlcHdlCrlcUlCfgReq
438 (

465  case CKW_CFG_DELETE:

470  &cfgTmpData->cfgEntData[idx].entUlCfgCfm,

532  }

537  rlcCfgApplyDelUlUe(gCb, cfgTmpData);
538  RLC_MEM_CPY(entCfgCfm,

541  }

546  }

579  cfgCfm->numEnt = cfg->numEnt;
580 
581  return ROK;
